server {
    server_name gsnowboarding.com www.gsnowboarding.com;
    root /var/www/gsnowboarding;
    index index.html;
    try_files $uri $uri/ =404;

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/gsnowboarding.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/gsnowboarding.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

    location ~ /\.git {
    deny all;
    return 404;
}
}


server {
    if ($host = www.gsnowboarding.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = gsnowboarding.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name gsnowboarding.com www.gsnowboarding.com;
    return 404; # managed by Certbot
