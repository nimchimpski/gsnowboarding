server {
    server_name nimchimpski.com www.nimchimpski.com;
    root /var/www/server-index;
    index index.html;
    try_files $uri $uri/ =404;
    location /ayou/ {
        proxy_pass http://unix:/var/www/ayou/ayou.sock;
        include proxy_params;
    
    location ~ /\.git {
    deny all;
    return 404;
    }
    }
    location /static/ {
        alias /var/www/ayou/staticfiles/;
    }
    location /nimchimpski_static/ {
        alias /var/www/server-index/;
    }
    location /ttt/ {
        proxy_pass http://127.0.0.1:5002;
        proxy_set_header SCRIPT_NAME /ttt;
        include proxy_params;
    }
    location /othello/ {
        proxy_pass http://127.0.0.1:5003;
        proxy_set_header SCRIPT_NAME /othello;
        include proxy_params;
    }
    location /gsnowboarding {
        root /var/www;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/nimchimpski.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/nimchimpski.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}
